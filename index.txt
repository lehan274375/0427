  <script>
    /**
     * 성공 시 UI 업데이트 처리
     */
    function handleSuccess(response) {
      const btn = document.getElementById('submitBtn');
      const btnText = document.getElementById('btnText');
      const btnIcon = document.getElementById('btnIcon');
      const btnSpinner = document.getElementById('btnSpinner');
      const statusMsg = document.getElementById('statusMessage');
      const feedbackCard = document.getElementById('feedbackCard');

      if (response.success) {
        statusMsg.classList.remove('hidden');
        feedbackCard.classList.remove('hidden');
        feedbackCard.classList.add('animate-up');
        document.getElementById('aiFeedback').innerText = response.feedback;
        
        btnText.innerText = "기록 완료!";
        btnSpinner.classList.add('hidden');
        btnIcon.classList.remove('hidden');
        btn.classList.replace('btn-submit', 'bg-slate-300');
        btn.style.boxShadow = "none";
        btn.style.transform = "none";
      } else {
        alert(response.error);
        resetBtn();
      }
    }

    /**
     * 전송 버튼 클릭 핸들러
     */
    function handleSubmit() {
      const name = document.getElementById('studentName').value.trim();
      const keywords = document.getElementById('keywords').value.trim();
      const reason = document.getElementById('reason').value.trim();
      
      if (!name || !keywords) {
        alert("이름과 오늘의 기분을 꼭 적어주세요! 😊");
        return;
      }

      // UI 로딩 상태
      const btn = document.getElementById('submitBtn');
      const btnText = document.getElementById('btnText');
      const btnIcon = document.getElementById('btnIcon');
      const btnSpinner = document.getElementById('btnSpinner');
      
      btn.disabled = true;
      btnText.innerText = "기록 중...";
      btnIcon.classList.add('hidden');
      btnSpinner.classList.remove('hidden');

      // Google Apps Script 연동
      if (typeof google !== 'undefined' && google.script) {
        google.script.run
          .withSuccessHandler(handleSuccess)
          .withFailureHandler((err) => {
            alert("연결 오류가 발생했어요. 다시 한번 시도해볼까요?");
            resetBtn();
          })
          .submitData(name, keywords, reason);
      } else {
        // 테스트용 시뮬레이션
        setTimeout(() => {
          handleSuccess({
            success: true, 
            feedback: "오늘 정말 많이 힘들었겠구나. 세 번이나 수행평가를 보느라 피곤하고 불안했을 너의 마음이 느껴져서 안타까워. 잘 못 본 것 같아 속상한 마음도 충분히 이해해. 하지만 네 노력이 소중하니, 조금 쉬면서 다시 힘내길 응원할게."
          });
        }, 1500);
      }
    }

    function resetBtn() {
      const btn = document.getElementById('submitBtn');
      const btnText = document.getElementById('btnText');
      const btnIcon = document.getElementById('btnIcon');
      const btnSpinner = document.getElementById('btnSpinner');
      btn.disabled = false;
      btnText.innerText = "감정 기록하기";
      btnIcon.classList.remove('hidden');
      btnSpinner.classList.add('hidden');
    }
  </script>
